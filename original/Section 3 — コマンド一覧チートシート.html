修正箇所 4つ
# Bug 1: import requests が無い
                                   → import requests を追加

# Bug 2: host が空文字
host = ""                          → host = "10.3.11.1"

# Bug 3: GET に不要な data=payload（payload未定義でNameError）

修正済み完全スクリプト
import json import requests import urllib3 urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning) host = "10.3.11.1" port = "443" url = f"https://{host}:{port}/restconf/data/Cisco-IOS-XE-interfaces-oper:interfaces/interface" headers = { "Accept": "application/yang-data+json", "Content-Type": "application/yang-data+json" } response = requests.request( 'get', url, auth=("netadmin", "CCIEinfra"), headers=headers, verify=False ) jsondata = response.json() print("-" * 100) print(f"{'Interface':<45} {'Ingress Octets':<30} {'Egress Octets':<30}") print("-" * 100) for interface in jsondata["Cisco-IOS-XE-interfaces-oper:interface"]: if (interface['oper-status'] == 'if-oper-state-ready' and interface['interface-type'] == 'iana-iftype-ethernet-csmacd'): switchport = interface["name"] ingressoctets = interface["statistics"]["in-octets"] egressoctets = interface["statistics"]["out-octets"] print(f"{switchport:<45} {ingressoctets:<30} {egressoctets:<30}")
